Wszystko to POST.
Pod routem jest lista parametrów, jakie należy przesłać jako obiekt json.

/graph/create : void
name	- string

/place/create : int id
label	- string
markers	- int
posx	- int
posy	- int

/transition/create : int id
label	- string
posx	- int
posy	- int

/place/markers : void
id	- int
markers	- int

/vertex/label : void
id	- int
label	- string

/vertex/position : void
id	- int
posx	- int
posy	- int

/vertex/remove : void
id	- int

/vertex/connect : int id
source	- int
target	- int

/vertex/disconnect : void
source	- int
target	- int
